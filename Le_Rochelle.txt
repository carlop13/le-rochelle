create database lerochelle;

use lerochelle;

create table usuario (
id int auto_increment not null, 
password varchar (50),
tipo varchar (20),
activo int,
token varchar (32),
constraint primary key(id));

desc usuario;

create table cliente(
id int auto_increment not null,
nombre varchar (50),
correo varchar (100),
fec_registro date,
tel varchar(10),
id_usu int,
constraint primary key(id),
constraint foreign key(id_usu) references usuario(id));

desc cliente;

create table propietario(
id int auto_increment not null,
nombre varchar (50),
ap varchar (30),
correo varchar (100),
fec_registro date,
tel varchar(10),
id_usu int,
constraint primary key(id),
constraint foreign key(id_usu) references usuario(id));

desc propietario;

create table administrador(
id int auto_increment not null,
nombre varchar (50),
correo varchar (100),
id_usu int,
constraint primary key(id),
constraint foreign key(id_usu) references usuario(id));

desc administrador;


create table datos (
id int auto_increment not null,
nombre VARCHAR(150),  
calle VARCHAR(255),
numero varchar(20),
colonia varchar(200),
cp varchar(20),
ciudad VARCHAR(100),
estado VARCHAR(100),
pais VARCHAR(100),
tel VARCHAR(15),
correo VARCHAR(200),
horario VARCHAR(30),
lat VARCHAR(30),
lng VARCHAR(30),
fb VARCHAR(120),
ig VARCHAR(120),
tk VARCHAR(120),
ws varchar(10),
constraint primary key(id));

desc datos;

create table ciudad(
id int auto_increment not null,
nombre VARCHAR(150),
id_pais int,

constraint primary key(id),
constraint foreign key(id_pais) references pais(id));

desc ciudad;

create table tipo(
id int auto_increment,
tipo varchar(50), 
activo int,
constraint primary key(id));

desc tipo;


create table moneda(
id int auto_increment,
nombre varchar(25), 
simbolo varchar(10), 
signo varchar(10) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci, 
constraint primary key(id));

desc moneda;


create table propiedad(
id int auto_increment,
titulo varchar(50),
calle varchar(80),
col varchar(80),
noext varchar(50),
habitaciones varchar(20),
banios varchar(20),
pisos varchar(20),  
garage varchar(5),
precio float(16,2),
foto varchar(180),
id_ciu int,
id_propi int,
id_tipo int,
estado varchar(20),
lat decimal(18,12),
lng decimal(18,12),

noesta varchar(20),
ancho varchar(60),
fondo varchar(60),
suptot varchar(60),

create table descripcion(
id int auto_increment,
descripcion varchar(900), 
id_prop int,
constraint primary key(id),
constraint foreign key(id_prop) references propiedad(id));

desc descripcion;

create table foto(
id int auto_increment,
foto varchar(180), 
id_prop int,
constraint primary key(id),
constraint foreign key(id_prop) references propiedad(id));

desc foto;

create table destacado(
id int auto_increment,
id_prop int,
constraint primary key(id),
constraint foreign key(id_prop) references propiedad(id));

desc destacado;

create table solicitud(
id int auto_increment,
id_cli int,
id_prop int,
mensaje varchar(180),
fecha date,
constraint primary key(id),
constraint foreign key(id_prop) references propiedad(id),
constraint foreign key(id_cli) references cliente(id));

desc solicitud;

create table resenia(
id int auto_increment,
id_cli int,
id_propietario int,
resenia varchar(180),
fecha datetime,
constraint primary key(id),
constraint foreign key(id_propietario) references propietario(id),
constraint foreign key(id_cli) references cliente(id));

desc resenia;

create table inquilino(
id int auto_increment,
id_cli int,
id_prop int,
fecha date,
constraint primary key(id),
constraint foreign key(id_prop) references propiedad(id),
constraint foreign key(id_cli) references cliente(id));

desc inquilino;

create table estrella(
id int auto_increment,
estrellas float(5,1),
fecha date,
id_cliente int,
id_propietario int,
constraint primary key(id),
constraint foreign key(id_propietario) references propietario(id),
constraint foreign key(id_cliente) references cliente(id));

desc estrella;


create table mensaje(
id int auto_increment,
nombre varchar(80),
correo varchar(180),
tel varchar(10),
fecha datetime,
mensaje varchar(150), 
constraint primary key(id));
